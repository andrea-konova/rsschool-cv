function getMatrix(number) {
  let matrix = [];
  for (let i = 0; i < number; i++) {
    matrix[i] = [];
    for (let j = 0; j < number; j++) {
      matrix[i][j] = (i === j ? 1 : 0);
    }
  }
  return matrix;
}
